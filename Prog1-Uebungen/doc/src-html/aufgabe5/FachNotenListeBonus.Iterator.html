001package aufgabe5;
002import java.util.NoSuchElementException;
003
004/**
005 * FachNotenListe beinhaltet Noten und ihre Fachnamen.
006 */
007public class FachNotenListeBonus {
008
009    private Element head = null; // leere Liste
010    /**
011     * F&uuml;gt eine Zahl am Listenanfang ein.
012     * Der heimliche Paramter this verweist auf das Objekt des Aufrufs.
013     * @param n die einzuf&uuml;gende Zahl
014     * @return die Liste
015     */
016    public FachNotenListeBonus insert(final FachNoteBonus n) {
017        this.head = new Element(this.head, n);
018        return this;
019    }
020
021    /**
022     * Element speichert eine einzelne Zahl als Teil einer Liste.
023     * Beipiel f&uuml;r eine statisch eingebettete Klasse.
024     */
025    private static final class Element {
026        private final Element next;
027        private final FachNoteBonus n;
028
029        private Element(/* final Element this, */
030                        final Element e, final FachNoteBonus n) {
031            this.next = e;
032            this.n = n;
033        }
034    }
035
036    /**
037     * Iterator speichert den aktuellen Zustand einer Listeniteration.
038     * Beipiel f&uuml;r eine innere Klasse.
039     */
040    public final class Iterator {
041        // private FachNotenListe FachNotenListe.this;
042        private Element current = FachNotenListeBonus.this.head;
043
044        /**
045         * pr&uuml;t, ob das Listenende erreicht ist.
046         * Der heimliche Paramter this verweist auf das Objekt des Aufrufs.
047         * @return true, wenn das Listenende erreicht ist, sonst false.
048         */
049        public boolean hasNext(/* final Iterator this */) {
050            return this.current != null;
051        }
052
053        /**
054         * liefert die aktuelle Zahl und iteriert zum n&auml;chsten Jahr.
055         * Aufruf am Listenende liefert NoSuchElementException.
056         * Der heimliche Paramter this verweist auf das Objekt des Aufrufs.
057         * @return die aktuelle Zahl
058         */
059        public FachNoteBonus next(/* final Iterator this */) {
060            if (this.current == null) {
061                throw new NoSuchElementException();
062            }
063
064            Element e = this.current;
065            this.current = this.current.next;
066            return e.n;
067        }
068    }
069}
