001package aufgabe5;
002
003/**
004 * Note Klasse.
005 */
006public final class NoteBonus {
007
008    /**
009     * Beste mögliche Note.
010     */
011    public static final NoteBonus BESTE = new NoteBonus(10);
012
013    /**
014     * Schlechteste mögliche Note.
015     */
016    public static final NoteBonus SCHLECHTESTE = new NoteBonus(50);
017
018    private int wert;
019
020    private NoteBonus(final int wert) {
021        super();
022
023        this.wert = wert;
024    }
025
026    /**
027     * Fabrikmethode zum erzeugen einer Note.
028     * @param n Interger zwischen 10 und 50
029     * @return Gibt ein neues Objekt von Note zurück;
030     */
031    public static NoteBonus valueOf(final int n) {
032
033        if (!istZulaessig(n)) {
034            throw new IllegalArgumentException("unzulaessige Note " + n);
035        }
036
037        return new NoteBonus(n);
038    }
039
040    /**
041     * Fabrikmethode zum erzeugen einer Note aus dem Mittelwert der beiden 
042     * uebergebenen Noten.
043     * @param n1 Objekt vom Typ Note.
044     * @param n2 Objekt vom Typ Note.
045     * @return Gibt ein neues Objekt von Note zurück;
046     */
047    public static NoteBonus valueOf(NoteBonus n1, NoteBonus n2) {
048        return new NoteBonus((n1.wert + n2.wert) / 2);
049    }
050
051
052    private static boolean istZulaessig(final int n) {
053        int[] valid = {10, 13, 17, 20, 23, 27, 30, 33, 37, 40, 50};
054
055        for (int var : valid) {
056            if (n == var) {
057                return true;
058            }
059        }
060        return false;
061    }
062
063    /**
064     * Gibt die Note zurück.
065     * @return Note als Integer.
066     */
067    public int intValue() {
068        return this.wert;
069    }
070
071    /**
072     * Ueberprueft ob Note zulaessig ist.
073     * @return True wenn zulaessig.
074     */
075    public boolean istBestanden() {
076        return this.wert <= 40;
077    }
078
079    @Override
080    public String toString() {
081        return String.format(this.wert / 10 + "," + this.wert % 10);
082    }
083
084    @Override
085    public boolean equals(final Object obj) {
086        return super.equals(obj);
087    }
088
089    @Override
090    public int hashCode() {
091        return this.intValue();
092    }
093}
