001package aufgabe6;
002/**
003 * Faecher.
004 */
005public final class Faecher {
006
007    private static final String[] FAECHER = {"Mathematik 1",
008                                             "Programmiertechnik 1",
009                                             "Digitaltechnik",
010                                             "Konsolidierung",
011                                             "FMSM",
012                                             "SM2"};
013
014    private Faecher() {
015        super();
016    }
017
018    /**
019     * Ueberprueft ob das uebergebene Fach in dem Array liegt.
020     * @param fach Fach
021     * @return true wenn zulaessig
022     */
023    public static boolean istZulaessig(final String fach) {
024        for (String s : FAECHER) {
025            if (s.equals(fach)) {
026                return true;
027            }
028        }
029        return false;
030    }
031
032}
