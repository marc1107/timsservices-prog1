001package aufgabe6;
002
003import aufgabe5.Note;
004
005/**
006 * BenoteteLeistung.
007 */
008public class BenoteteLeistung extends Leistung {
009
010    private final Note note;
011
012    /**
013     * Konstruktor.
014     * @param fach Fach
015     * @param note Note
016     */
017    public BenoteteLeistung(final String fach, final String note) {
018        super(fach);
019        this.note = Note.valueOf(Integer.parseInt(note));
020    }
021
022    @Override
023    public String getNote() {
024        if (istBenotet()) {
025            return this.note.toString();
026        }
027        return "";
028    }
029
030    @Override
031    public String getNoteInWorten() {
032        if (istBenotet()) {
033            if (istBestanden()) {
034                if (note.intValue() >= 10 && note.intValue() <= 15) {
035                    return "sehr gut";
036                } else if (note.intValue() >= 16 && note.intValue() <= 25) {
037                    return "gut";
038                } else if (note.intValue() >= 26 && note.intValue() <= 35) {
039                    return "befriedigend";
040                } else if (note.intValue() >= 36 && note.intValue() <= 40) {
041                    return "ausreichend";
042                } else {
043                    return "nicht ausreichend";
044                }
045            } else {
046                return "nicht bestanden";
047            }
048        }
049        return "";
050    }
051
052    @Override
053    public boolean istBenotet() {
054        return true;
055    }
056
057    @Override
058    public boolean istBestanden() {
059        if (this.note.istBestanden()) {
060            return true;
061        }
062        return false;
063    }
064}
