001package aufgabe6;
002
003import java.io.PrintWriter;
004
005/**
006 * HtmlFormat.
007 */
008public final class HtmlFormat {
009
010    private HtmlFormat() { }
011
012    /**
013     * Schreibt die Ausgabe in eine HTML-Datei.
014     * @param pw PrintWrite Object
015     * @param name Name des Schuelers
016     * @param fnl FachNotenListe Object
017     */
018    public static void ausgeben(final PrintWriter pw, final String[] name,
019        final FachNotenListe fnl) {
020        pw.println("<!DOCTYPE html>");
021        pw.println("<html lang=\"de\">");
022        pw.println("<head>");
023        pw.println("<meta charset=\"utf-8\">");
024        pw.println("<meta name=\"viewport\" content=\"width=device-width, "
025            + " initial-scale=1.0\">");
026        pw.println("<style>");
027        pw.println("table { width:100%; }");
028        pw.println("th { text-align:left; }");
029        pw.println("</style>");
030        pw.println("<title>Notenspiegel</title>");
031        pw.println("</head>");
032        pw.println("<body>");
033        pw.println("<h1>Notenspiegel f&uuml;r");
034        pw.println(name[0]);
035        pw.println(name[1]);
036        pw.println("</h1>");
037        pw.println("<hr>");
038        pw.println("<table>");
039        pw.println("<tr><th>Fach:</th><th>Art:</th><th>Note:</th></tr>");
040
041        for (Leistung var : fnl) {
042
043            if (var.istBestanden()) {
044                if (var.istBenotet()) {
045                    pw.printf("<tr><td>%s</td><td>L</td><td>%s (%s)</td>"
046                        + "</tr>%n",
047                            var.getFach(), var.getNoteInWorten(),
048                                var.getNote());
049                } else {
050                    pw.printf("<tr><td>%s</td><td>S</td><td>%s</td></tr>%n",
051                        var.getFach(), var.getNoteInWorten());
052                }
053            } else {
054                pw.printf("<tr><td style=\"color:red\">%s</td><td>S</td><td>"
055                    + "nicht bestanden</td></tr>", var.getFach());
056            }
057        }
058
059        pw.println("</table>");
060        pw.println("<hr>");
061        pw.println("L = Leistungsnachweis, S = Schein");
062        pw.println("</body>");
063        pw.println("</html>");
064    }
065}
