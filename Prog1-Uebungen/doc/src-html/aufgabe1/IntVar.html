003package aufgabe1;
004
005import java.util.Scanner;
006
007/**
008 * IntVar zeigt den Umgang mit Variablen vom Typ int.
009 * %Uuml;bungsaufgabe 1 zur Programmiertechnik 1.
010 * @author Tim Koehler
011 * @version 22.10.18
012 */
013public final class IntVar {
014    private IntVar() { }
015
016    private static final Scanner EINGABE = new Scanner(System.in);
017
018    /**
019     * main ist der Startpunkt des Programms.
020     * @param args wird nicht verwendet.
021     */
022    public static void main(final String[] args) {
023
024        /* Konstanten min und max definieren */
025        final int min = Integer.MIN_VALUE;
026        final int max = Integer.MAX_VALUE;
027
028
029        /* Eingabeaufforderung ausgeben */
030        System.out.printf("Zwei ganze Zahlen zwischen %d"
031            + " und %d eingeben: %n", min, max);
032
033
034        /* zwei ganze Zahlen einlesen */
035        long num1 = 0;
036        long num2 = 0;
037        long ergebnis = 0;
038
039        if (EINGABE.hasNextInt()) {
040            num1 = EINGABE.nextInt();
041        } else {
042            System.out.println("Fehlerhafte eingabe!");
043            return;
044        }
045
046        if (EINGABE.hasNextInt()) {
047            num2 = EINGABE.nextInt();
048        } else {
049            System.out.println("Fehlerhafte eingabe!");
050            return;
051        }
052
053        /* die beiden Zahlen dezimal, okatal und
054        hexadezimal ausgeben */
055        System.out.printf("%d ist oktal %o und hexadezimal %x%n",
056            num1, num1, num1);
057        System.out.printf("%d ist oktal %o und hexadezimal %x%n",
058            num2, num2, num2);
059
060        System.out.print("Operator eingeben: ");
061        String operator = EINGABE.next();
062
063        switch (operator) {
064        case "+":
065            ergebnis = num1 + num2;
066            break;
067        case "-":
068            ergebnis = num1 - num2;
069            break;
070        case "*":
071            ergebnis = num1 * num2;
072            break;
073        case "/":
074            ergebnis = num1 / num2;
075            break;
076        case "%":
077            ergebnis = num1 % num2;
078            break;
079        case "==":
080            System.out.printf("%d == %d ist %b%n", num1, num2, num1 == num2);
081            break;
082        case "!=":
083            System.out.printf("%d != %d ist %b%n", num1, num2, num1 != num2);
084            break;
085        case "<":
086            System.out.printf("%d < %d ist %b%n", num1, num2, num1 < num2);
087            break;
088        case "<=":
089            System.out.printf("%d <= %d ist %b%n", num1, num2, num1 <= num2);
090            break;
091        case ">":
092            System.out.printf("%d > %d ist %b%n", num1, num2, num1 > num2);
093            break;
094        case ">=":
095            System.out.printf("%d >= %d ist %b%n", num1, num2, num1 >= num2);
096            break;
097        default:
098            System.out.println("Fehlerhafte Eingabe!");
099            return;
100        }
101
102        if (ergebnis != 0) {
103            if (ergebnis > Integer.MAX_VALUE || ergebnis < Integer.MIN_VALUE) {
104                System.out.println("Ergebnis zu gro√ü oder zu "
105                    + "klein um als Integer dargestelt zu werden!");
106            } else {
107                if (!operator.equals("%")) {
108                    System.out.printf("%d " + operator + " %d ist %d%n",
109                        num1, num2, ergebnis);
110                } else {
111                    System.out.printf("%d %% %d ist %d%n",
112                        num1, num2, ergebnis);
113                }
114            }
115        }
116    }
117}
